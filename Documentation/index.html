Rappix - App Landing Page

Rappix - App Landing Page is a responsive App landing page designed for app showcase. Anyone can use this template showcase any product and customize as his wishes.

  • created: 05/05/2020
  • latest update: 05/05/2020
  • by: freekytheme

02.Erase previous Mailchimp Form code from script.js & add this code from updated template & Replace url with your own mailchimp post URL

/* ==========================================================================
      Mailchimp ajax
    ========================================================================== */
    if($('.mailchimp').length > 0) {
        /*  MAILCHIMP  */
        $('.mailchimp').ajaxChimp({
            language: 'es',
            callback: mailchimpCallback,
            url: "https://gmail.us8.list-manage.com/subscribe/post?u=dd00a6e222a827748e67dd5d6&amp;id=5f045a38b3" //Replace this with your own mailchimp post URL. Don't remove the "". Just paste the url inside "".
        });
    }

    function mailchimpCallback(resp) {
        if (resp.result === 'success') {
            $('.subscription-success').html(resp.msg).fadeIn(1000);
            $('.subscription-error').fadeOut(500);

        } else if (resp.result === 'error') {
            $('.subscription-error').html(resp.msg).fadeIn(1000);
        }
    }
    $.ajaxChimp.translations.es = {
        'submit': 'Submitting...',
        0: 'We have sent you a confirmation email',
        1: 'Please enter a value',
        2: 'An email address must contain a single @',
        3: 'The domain portion of the email address is invalid (the portion after the @: )',
        4: 'The username portion of the email address is invalid (the portion before the @: )',
        5: 'This email address looks fake or invalid. Please enter a real email address'
    };
